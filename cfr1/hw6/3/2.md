-- (∀ α, β, γ : Type)(∃ f : (γ^β)^α → γ^(βα))(∃ f' : γ^(βα) → (γ^β)^α)[f ∘ f' = id & f' ∘ f = id]

f : (γ^β)^α → γ^(βα) -- (α → β → γ) → β × α → γ
f g = λw.(g w·r) w·l

f' : γ^(βα) → (γ^β)^α -- (β × α → γ) → α → β → γ
f' g = λa.λb. g (b, a)
