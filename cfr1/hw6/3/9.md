-- δ(α + β) ≅ δα + δβ
-- (∀ α, β, δ : Type)(∃ f : δ(α + β) → δα + δβ)(∃ f' : δα + δβ → δ(α + β))[f ∘ f' = id & f' ∘ f = id]

f : δ(α + β) → δα + δβ
f (d, l·a) = l·(d, a)
f (d, r·b) = r·(d, b)

f' : δα + δβ → δ(α + β)
f' l·(d, a) = (d, l·a)
f' r·(d, b) = (d, r·b)

Parte L:
  Seja s : δα + δβ.
  Separo em casos de s.
  Caso l·(d, a):
    Calc:
      (f ∘ f') l·(d, a)
      = f (f' l·(d, a))
      = f (d, l·a)
      = l·(d, a)
      = id l·(d, a)
  Caso r·(d, b):
    Similar.

Parte R:

  Separo em casos de w:
    Caso (d, l·a)
      Calc:
        (f' ∘ f) (d, l·a)
        = f' (f (d, l·a))
        = f' l·(d, a)
        = (d, l·a)

    Caso (d, r·b):
      Similar.
