-- 𝟘^𝟘 ≅ 𝟙
-- (∀ 𝟘 : Type)(∃ f : 𝟘^𝟘 → 𝟙)(∃ f' : 𝟙 → 𝟘^𝟘)[f ∘ f' = id & f' ∘ f = id]

f : 𝟘^𝟘 → 𝟙 -- (𝟘 → 𝟘) → 𝟙
f g = ☆

f' : 𝟙 → 𝟘^𝟘 -- 𝟙 → (𝟘 → 𝟘)
f' ☆ = id

Parte L:
  Calc:
    (f ∘ f') ☆
    = f (f' ☆)
    = f (id)
    = ☆
    = id ☆

Parte R:
  Seja g : 𝟘 → 𝟘.
  Calc:
    (f' ∘ f) g
    = f' (f g)
    = f' ☆
    = id
    = g [(→ 𝟘)]
