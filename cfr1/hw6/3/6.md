-- 𝟙^α ≅ 𝟙
-- (∀ α : Type)(∃ f : 𝟙^α → 𝟙)(∃ f' : 𝟙 → 𝟙^α)[f ∘ f' = id & f' ∘ f = id]

f : 𝟙^α → 𝟙 -- (α → 𝟙) → 𝟙
f g = ☆

f' : 𝟙 → 𝟙^α -- 𝟙 → (α → 𝟙)
f' ☆ = λa.☆

Parte L:
  Calc:
    (f ∘ f') ☆
    = f (f' ☆)
    = f (λa.☆)
    = ☆
    = id ☆

    (f' ∘ f) g
    = f' (f g)
    = f' ☆
    = λa.☆
    = g [(→ 𝟙)]
    = id g
