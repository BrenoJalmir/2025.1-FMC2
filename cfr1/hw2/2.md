```
-- (∃A : Set α)[Empty A]
Seja A = {a : α | a ≠ a}
Seja x : α.
Logo x ∉ A [x=x].
Escolho A.

-- (∀A, B : Set α)[Empty A ∧ Empty B ⇒ A = B]

Seja h : Empty A ∧ Empty B.
Parte (∀x : α)[x ∈ A  ⇒ x ∈ B]
  Seja y ∈ A.
  Contradição [h·l; Empty.1]
Parte (∀x : α)[x ∈ B ⇒ x ∈ A]
  Seja y ∈ B.
  Contradição [h·r; Empty.1]
```
